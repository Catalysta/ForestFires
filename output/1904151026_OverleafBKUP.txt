\frame{
\frametitle{Introduction}
\begin{itemize}
  \item Can weather variables be used to predict the size of forest fires?
 \vskip 0.5cm
   \item How can machine learning and data mining leverage readily available cheap meteorological data to mitigate or prevent disaster?
\vskip 0.5cm
\item Burn area predictions could possibly inform resource allocation (e.g., fire trucks versus helicopters) and preparedness.
\end{itemize}
\begin{figure}[fig]
\centering
\includegraphics[width=4cm]{Images/Wild-fires.jpg}
\caption{\tiny{Wildfire near Fawnskin, California. Photo: David McNew / Getty Images}}
\label{fig:wildfire}
\end{figure}
} 


\frame{
\frametitle{Data}
\begin{itemize}
  \item Data from Paulo Paulo Cortez and Anibal Morais at the University of Minho, in Guimaraes, Portugal
  \item 288 fires in the Montesinho National Park in northeast Portugal between January 2000 and December 2003.
  \item Twelve predictors to model the total burned area of large forest fires within the park.
\item 
\begin{enumerate}
\item \textbf{X}: x-axis spatial coordinate within the Montesinho park map: 1 to 9
\item \textbf{Y}: y-axis spatial coordinate within the Montesinho park map: 2 to 9
\item \textbf{month}: month of the year: ``jan'' to ``dec''
\item \textbf{day}: day of the week: ``mon'' to ``sun''
\item \textbf{FFMC}: index from the FWI system: 18.7 to 96.20
\end{enumerate} 
\end{itemize}
}


\frame{
\frametitle{Data}
\begin{itemize}
  \item \begin{enumerate}
        \setcounter{enumi}{5}
        \item \textbf{DMC} index from the FWI system: 1.1 to 291.3
        \item \textbf{DC} index from the FWI system: 7.9 to 860.6
        \item \textbf{ISI} index from the FWI system: 0.0 to 56.10
        \item \textbf{temp}: temperature in Celsius degrees: 2.2 to 33.30
        \item \textbf{RH}: relative humidity in \%: 15.0 to 100 
        \item \textbf{wind}: wind speed in km/h: 0.40 to 9.40
        \item \textbf{rain} outside rain in mm/m2 : 0.0 to 6.4\\
    \end{enumerate}
    \item \textit{Response variable}
    \begin{itemize}
        \item \textbf{area}: the burned area of the forest (in ha): 0.00 to 1090.84
    \end{itemize}
    \item Fires that burn less than $100m^2$ of area are all catalogued as zero. 
    \item There are only 7 nonzero \textbf{rain} values: Considered \textbf{rainbin}
\end{itemize}
}


\frame{
\frametitle{Methods: Multiple Logistic Regression (MLR)}
\begin{itemize}
  \item This approach uses the \textbf{area} variable with a log transformation
  \item Out of 517 records, there are 247 with an area value of 0.
  \begin{itemize}
      \item Missing data!
  \end{itemize}
  \item We can apply the MLR model to the data conditioned to nonzero burn area only.
\end{itemize}

\frame{
\frametitle{Methods: MLR}
\begin{itemize}
  \item   Nonzero burn area data presents with high multicollinearity, no clear predictors even after transformation.
  \begin{figure}[fig]
  \centering
  \includegraphics[width=5cm]{Plots/PairsPlot.jpg}
  \end{figure}
  \item Noisy process with potentially non-constant variance.
  \item Response time, method of containment important, but missing.
\end{itemize}
}

\frame{
\frametitle{MLR Results}
\begin{itemize}
  \item Standard methods of analysis not very fruitful!
    \begin{figure}[fig]
    \centering
    \includegraphics[width=6cm, height=2cm]{Plots/RegSubsets.jpg}
    \end{figure}
  \item Instead consider all quadratic, cubic, and two-way, three-way interaction terms as potential predictors.
  \item From these, LASSO suggests  17-variable model with mostly cubic and interaction terms; only two of original predictors. (in-sample MSE $\approx$2)
  \item Backward Selection suggests 40-variable model with mostly interaction terms (Adj. $R^2$=0.199, in-sample MSE=1.47)
\end{itemize}
}

\frame{
\frametitle{Methods: Logistic Regression}
\begin{itemize}
  \item We considered two types of outcomes:
    \begin{itemize}
        \item Small fires (less than $100m^2$ of burn area)
        \item Large fires (more than $100m^2$ of burn area)
    \end{itemize}
  \item We compared the deviances of all the $2^p - 1$ possible models to perform variable selection
  \item Only weather-related variables were considered.
\end{itemize}
\begin{figure}[fig]
\centering

\frame{
\frametitle{Discussion}
\begin{itemize}
  \item Residual plots for MLR model actually suggest nonlinear relationship: polynomial or splines. 
    \begin{figure}[fig]
    \centering
    \includegraphics[width=6.5cm]{Plots/ResidsMLR.jpg}
\end{figure}
  \item Assumptions are not met here, so any result of this analysis should not be used to actually fight fires.
  \item Instead it looks like there is potential to use weather variables to predict size of fire, but other data is also needed, such as weather data on non-fire days
  \item DMC, RH, TrFFMC, and wind were important in the MLR and LASSO models. Wind also appears in the logistic model
\end{itemize}
}
